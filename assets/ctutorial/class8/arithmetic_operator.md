### Arithmetic Operators

![](resource:assets/images/C/img-b7.png)

```
#include<stdio.h>
int main()
{
	int a=5,b=3; 	//For Operation
	int add,sub,multi,div,mod; 	//To store the result
	add=x+y;
	sub=x-y;
	multi=x*y;
	div=x/y;
	mod=x%y;
	printf("Addition=%d\n",add);
	printf("Subtraction=%d\n",sub);
	printf("Multiplication=%d\n",multi);
	printf("Division=%d\n",div);
	printf("Modulus=%d\n",mod); 	// %(modulus) holds remainder
}

/*

Addition=8
Subtraction=2
Multiplication=15
Division=1
```
